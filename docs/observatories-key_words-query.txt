SELECT observatories.id, observatories.name, targets.id, targets.target_name, research_areas.id, research_areas.name FROM observatories, research_areas, observatory_to_research_areas, targets, observatory_to_targets WHERE observatory_to_research_areas.observatory_id = observatories.id AND observatory_to_targets.observatory_id = observatories.id AND observatory_to_targets.target_id = targets.id AND observatory_to_research_areas.research_area_id = research_areas.id;
