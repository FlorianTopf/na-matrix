SELECT space_missions.id, space_missions.mission_name, targets.id, targets.target_family, targets.target_name, research_areas.id, research_areas.name, sensors.id, sensors.sensor_name, sensors.sensor_type, science_goals.id, science_goals.name FROM space_missions, targets, space_mission_to_targets, research_areas, sensors, science_goals, space_mission_to_research_areas, space_mission_to_sensors, sensor_to_science_goals WHERE space_missions.id = space_mission_to_targets.space_mission_id AND targets.id = space_mission_to_targets.target_id AND space_missions.id = space_mission_to_research_areas.space_mission_id AND research_areas.id = space_mission_to_research_areas.research_area_id AND sensors.id = sensor_to_science_goals.sensor_id AND science_goals.id = sensor_to_science_goals.science_goal_id AND space_missions.id = space_mission_to_sensors.space_mission_id AND sensors.id = space_mission_to_sensors.sensor_id;
